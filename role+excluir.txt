async function verificaRole(req, res, next) {
    const { role: userRole } = req.session.user;
    const role = 'admin';

    if (userRole !== role) {
        return res.status(403).json({ error: 'Você não tem permissão para acessar esta rota' });
    }

    next();
}

    }
